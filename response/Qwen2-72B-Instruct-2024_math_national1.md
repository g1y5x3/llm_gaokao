\[ \cos(\alpha - \beta) = \cos\alpha\cos\beta + \sin\alpha\sin\beta \]

\[ \tan\alpha = \frac{\sin\alpha}{\cos\alpha} \]

\[ m = \cos\alpha\cos\beta - 2\cos\alpha\cos\beta \]

\[ \cos(\alpha - \beta) = \cos\alpha\cos\beta + \sin\alpha\sin\beta \]

\[ \cos(\alpha - \beta) = 3\cos\alpha\cos\beta \]
