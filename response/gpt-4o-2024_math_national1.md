\[ a_1, a_1 + d, a_1 + 2d, \ldots, a_1 + (4m+1)d \]

\[ a_1, a_1 + 2d, a_1 + 3d, \ldots, a_1 + 11d, a_1 + 13d, \ldots, a_1 + (4m+1)d \]

\[ (a_1 + kd, a_1 + (k+1)d, a_1 + (k+2)d, a_1 + (k+3)d) \]
