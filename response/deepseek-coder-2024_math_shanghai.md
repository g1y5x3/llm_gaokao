\]

\[
\frac{2}{z} = \frac{2}{1 + b^2} - \frac{2b}{1 + b^2}i

z + \frac{2}{z} = (1 + bi) + \left( \frac{2}{1 + b^2} - \frac{2b}{1 + b^2}i \right)
\]

\]
